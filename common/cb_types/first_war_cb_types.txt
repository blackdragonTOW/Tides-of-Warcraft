tow_horde_invasion = {
	name = CB_NAME_TOW_HORDE_INVASION
	war_name = WAR_NAME_TOW_HORDE_INVASION
	sprite = 7
	truce_days = 3650
	is_permanent = yes
	hostile_against_others = yes
	full_hostility = yes
	display_on_map = yes
	allowed_to_target_tributaries = no
	defender_unoccupied_warscore = yes
	can_call_vassals = yes
	can_ask_to_join_war = yes
	allow_whitepeace = no
	can_call_allies = yes

	battle_warscore_mult = 0.75
	att_ticking_war_score_multiplier = 2
	def_ticking_war_score_multiplier = 0.25
	max_defender_occupation_score = 75
	max_defender_battle_score = 75
	capturing_defender_is_complete_victory = no
	capturing_attacker_is_complete_victory = no

	sort_priority = 2000
	
	can_use_gui = {
	}
	
	can_use = {
		NOT = { has_global_flag = TOW_first_war_started }
		has_global_flag = TOW_first_war_can_start
		year = 10000
		NOT = { year = 10005 }

		ROOT = {
			is_nomadic = no
			independent = yes
			mercenary = no
			OR = {
				is_tributary = no
				is_tributary = {
					type = offmap
				}
			}
			NOT = { is_vassal_or_below = FROM }
			NOT = { is_liege_or_above = FROM }
			NOT = {
				rightful_religious_head_scope = {
					character = FROM
				}
			}
			NAND = {
				holy_order = yes
				religion = FROM
			}
		}
		FROM = {
			independent = yes
			OR = {
				is_tributary = no
				is_tributary = {
					type = offmap
				}
			}
		}
		e_horde = {
			holder_scope = {
				character = ROOT
			}
		}
		k_stormwind = {
			holder_scope = {
				character = FROM
			}
		}
		
		# The attacker needs a border, or be at most two sea zones away from one of the target's counties
		FROM = {
			any_realm_province = {
				any_neighbor_province = {
					OR = {
						AND = {
							has_owner = yes
							owner = {
								OR = {
									character = ROOT
									is_liege_or_above = ROOT
									top_liege = {
										is_tributary = {
											type = permanent
											suzerain = ROOT
										}
									}
								}
							}
						}
						AND = {
							is_land = no
							any_neighbor_province = {
								OR = {
									AND = {
										has_owner = yes
										owner = {
											OR = {
												character = ROOT
												is_liege_or_above = ROOT
											}
										}
									}
									AND = {
										is_land = no
										any_neighbor_province = {
											owner = {
												OR = {
													character = ROOT
													is_liege_or_above = ROOT
												}
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}
	}

	is_valid = {
		NOT = {
			ROOT = {
				rightful_religious_head_scope = {
					character = FROM 
				}
			}
		}
	}

	on_add = {
		ROOT = { sound_effect = tow_first_war_orc }
		set_global_flag = TOW_first_war_started
		FROM = { narrative_event = { id = TOW_FW.17 } }
	}

	on_success = {
		ROOT = {
			show_scope_change = no
			occupy_minors_of_occupied_settlements = FROM
			gain_all_occupied_titles = { who = FROM type = invasion }
			prestige = 2500
		}

		FROM = {
			show_scope_change = no
			prestige = -2500
			any_demesne_title = {
				limit = {
					higher_tier_than = DUKE
					TOW_is_destructible_title = yes
				}
				destroy_landed_title = THIS
			}
		}
		
		if  = {
			limit = {
				ROOT = {
					uses_decadence = yes
				}
			}
			ROOT = {
				decadence = -10
			}
		}
		FROM = { disband_event_forces = first_war_troops }
	}

	on_fail = {
		ROOT = {
			show_scope_change = no
			prestige = -1000
		}
		FROM = {
			show_scope_change = no
			prestige = 1000
		}
		FROM = { disband_event_forces = first_war_troops }
	}

	on_reverse_demand = {
		ROOT = {
			show_scope_change = no
			prestige = -2000
			transfer_scaled_wealth = {
				to = FROM
				value = 5.0
			}
		}

		FROM = {
			show_scope_change = no
			prestige = 2000
		}
		FROM = { disband_event_forces = first_war_troops }
	}

	on_invalidation = {
		FROM = { disband_event_forces = first_war_troops }
	}
	
	on_attacker_leader_death = {
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}

	ai_will_do = {
		factor = 0.25
	}
}

	name = CB_NAME_PREEMPTIVE_STRIKE
	war_name = NAME_PREEMPTIVE_STRIKE
	sprite = 8
	truce_days = 3650
	is_permanent = yes
	hostile_against_others = yes
	can_ask_to_join_war = no
	display_on_map = no
	allowed_to_target_tributaries = no
	
	defender_unoccupied_warscore = yes
	
	sort_priority = 1000
	
	can_use = {
		always = no # only by script
	}

	can_use_gui = {
	}

	is_valid = {
		ROOT = {
			mercenary = no

			NOT = { is_vassal_or_below = FROM }
			NOT = { is_liege_or_above = FROM }
			NAND = {
				holy_order = yes
				religion = FROM
			}
			independent = yes
		}
		FROM = {
			NAND = {
				holy_order = yes
				religion = ROOT
			}
			independent = yes
		}
	}

	on_success = {
		FROM = {
			show_scope_change = no
			custom_tooltip = {
				text = stormwind_preemptive_strike_win_tooltip
				hidden_tooltip = {
					any_vassal = {
						limit = {
							higher_tier_than = COUNT
						}
						set_defacto_liege = THIS
					}
					if = {
						limit = { tier = EMPEROR }
						any_demesne_title = {
							limit = { tier = EMPEROR TOW_is_destructible_title = yes }
							if = {
								limit = { is_titular = yes }
		               			activate_title = { title = THIS status = no }
							}
							destroy_landed_title = THIS
							add_claim = FROM
							if = {
								limit = { is_titular = yes }
		               			activate_title = { title = THIS status = yes }
							}
						}
					}
					any_demesne_title = {
						limit = { tier = KING TOW_is_destructible_title = yes }
						if = {
							limit = { is_titular = yes }
		           			activate_title = { title = THIS status = no }
						}
						destroy_landed_title = THIS
						add_claim = FROM
						if = {
							limit = { is_titular = yes }
		           			activate_title = { title = THIS status = yes }
						}
					}
				}
			}
			imprison = ROOT
		}
		ROOT = {
			show_scope_change = no
			participation_scaled_prestige = 750
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { participation_scaled_prestige = 400 }
		}
	}

	on_fail = {
		ROOT = {
			prestige = -500
		}

		FROM = {
			show_scope_change = no
			participation_scaled_prestige = 500
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 250 }
		}
	}

	on_reverse_demand = {
		ROOT = {
			show_scope_change = no
			transfer_scaled_wealth = {
				to = FROM
				value = 5.0
				min = 200
			}
			prestige = -750
		}

		FROM = {
			show_scope_change = no
			participation_scaled_prestige = 600
		}

		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 300 }
		}
	}
	
	on_attacker_leader_death = {
		#end_war = invalid
	}

	attacker_ai_victory_worth = {

	}
