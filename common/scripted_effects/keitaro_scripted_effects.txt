TOW_employed_remove_unwanted_traits_with_ET_employer = {
	if = {
		limit = { event_target:employer = { accepts_warlock = no } }
		remove_trait = warlock
		remove_trait = necrolyte
	}
	if = {
		limit = { event_target:employer = { accepts_necromancer = no } }
		remove_trait = necromancer
		remove_trait = necrolyte
	}
	if = {
		limit = { event_target:employer = { accepts_assassin = no } }
		remove_trait = assassin
	}
	if = {
		limit = { event_target:employer = { accepts_rogue = no } }
		remove_trait = rogue
	}
	remove_trait = demon_hunter
	remove_trait = death_knight
}

# Create a courtier, can generate a female if allowed
# save primary title of employer as employer
# save employer as employer_character
TOW_create_random_courtier_gender_equality = {
	if = {
		limit = { gender_equality_trigger = no }
		create_character = {
			random_traits = yes
			dynasty = random
			female = no
			employer = event_target:employer
		}
	}
	if = {
		limit = { gender_equality_trigger = yes }
		random_list = {
			1 = {
				create_character = {
					random_traits = yes
					dynasty = random
					female = yes
					employer = event_target:employer
				}
			}
			1 = {
				create_character = {
					random_traits = yes
					dynasty = random
					female = no
					employer = event_target:employer
				}
			}
		}
	}
	new_character = {
		TOW_initialize_character_race_from_culture = yes
		TOW_update_character_race_from_liege = yes
	}
}

TOW_adventurers_spawn_unit_with_manpower = { 
	set_variable = { which = TOW_TMP which = TOW_adventure_manpower }
	while = {
		limit = {
			check_variable = {
				which = TOW_TMP
				value = 5
			}
		}
	
		random_list = { # 1/2 chance of disappearing after peace
			5 = {
				spawn_unit = {
					province = PREVPREV
					home = PREVPREV
					owner = THIS
					match_character = PREV
					match_mult = 0.05
					match_min = 125
					attrition = 0.25
					disband_on_peace = yes
					earmark = claimant_adventure
					merge = yes
				}
			}
			5 = {
				spawn_unit = {
					province = PREVPREV
					home = PREVPREV
					owner = THIS
					match_character = PREV
					match_mult = 0.05
					match_min = 125
					attrition = 0.25
					earmark = claimant_adventure
					merge = yes
				}
			}
		}
		subtract_variable = { which = TOW_TMP value = 5 }
	}
	while = {
		limit = {
			check_variable = {
				which = TOW_TMP
				value = 1
			}
		}
		spawn_unit = {
			province = PREVPREV
			home = PREVPREV
			owner = THIS
			match_character = PREV
			match_mult = 0.01
			match_min = 25
			attrition = 0.25
			disband_on_peace = yes
			earmark = claimant_adventure
			merge = yes
		}
		subtract_variable = { which = TOW_TMP value = 1 }
	}
	while = {
		limit = {
			check_variable = {
				which = TOW_TMP
				value = 0.1
			}
		}
		spawn_unit = {
			province = PREVPREV
			home = PREVPREV
			owner = THIS
			match_character = PREV
			match_mult = 0.001
			match_min = 2
			attrition = 0.25
			disband_on_peace = yes
			earmark = claimant_adventure
			merge = yes
		}
		subtract_variable = { which = TOW_TMP value = 0.1 }
	}
}

TOW_adventurers_spawn_fleet_with_manpower = { 
	export_to_variable = { which = TOW_TMP2 value = realm_levies who = event_target:adventure_target }
	multiply_variable = { which = TOW_TMP2 which = TOW_adventure_manpower }
	divide_variable = { which = TOW_TMP2 value = 10000 } # percentage * 100 soldier per fleet
	change_variable = { which = TOW_TMP2 value = 10 } # base fleet count

	set_variable = { which = TOW_TMP which = TOW_TMP2 }
	while = {
		limit = {
			check_variable = {
				which = TOW_TMP
				value = 50
			}
		}
	
		spawn_fleet = {
			province = closest # closest sea zone
			owner = ROOT
			disband_on_peace = yes
			troops =
			{
				galleys = { 100 100 }
			}
			earmark = claimant_adventure
		}
		subtract_variable = { which = TOW_TMP value = 50 }
	}
	while = {
		limit = {
			check_variable = {
				which = TOW_TMP
				value = 10

TOW_clear_landless_claimant_adventure = {
	hidden_tooltip = {
		clr_character_flag = TOW_preparing_adventure
		clear_flags_with_prefix = TOW_adventure_asked_
		set_variable = { which = "TOW_adventure_manpower" value = 0 }
		set_variable = { which = "TOW_adventure_realm_asked" value = 0 }
		remove_character_modifier = TOW_adventurer_puny_army

		remove_character_modifier = TOW_adventurer_powerful_army
		remove_character_modifier = TOW_adventurer_very_powerful_army
	}
}

TOW_landless_claimant_adventurer_update_modifier = {
	hidden_tooltip = {
		if = {
			limit = {
				check_variable = { which = "TOW_adventure_manpower" value = 15 }
				NOT = { check_variable = { which = "TOW_adventure_manpower" value = 40 } }
				NOT = { has_character_modifier = TOW_adventurer_puny_army }
			}
			add_character_modifier = {

				duration = -1
			}
		}
		if = {
			limit = {

				NOT = { check_variable = { which = "TOW_adventure_manpower" value = 65 } }
