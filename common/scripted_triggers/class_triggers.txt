	OR = {
		has_game_rule = {
			name = class_opinions
			value = off
		}
		trait = undead_generic
		trait = undead_scourge
		trait = undead_forsaken
		trait = naga
		trait = troll_forest
		trait = troll_ice
		trait = troll_jungle
		trait = troll_sand
		trait = troll_dark
		trait = troll_zandalari
		trait = eredar
		trait = succubus
		trait = dreadlord
		trait = fel_lord
		trait = shivarra
		trait = pit_lord
		trait = doomlord
		trait = voidwalker
		trait = faceless
		trait = hakkar_trait
		trait = ogre
		trait = wretched
		trait = blood_elf
		trait = felsworn
		religion_group = fel_group
		religion_group = void_group
		religion_group = wild_gods_group
		religion_group = goblin_group
		religion = cult_of_the_destroyer
		religion = cult_of_the_infinite
		religion = emerald_nightmare
		AND = {
			trait = warlock
			OR = {
				NOT = { has_character_flag = secret_warlock }
				has_character_flag = warlock_discovered
			}
		}
		AND = {
			trait = necrolyte
			OR = {
				NOT = { trait = gnoll }
				NOT = { has_character_flag = secret_necrolyte }
				has_character_flag = necrolyte_discovered
			}
		}
	}
}
