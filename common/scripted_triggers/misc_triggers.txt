quel_thalas_antimagic = {
	OR = {
		NOT = {
			any_province = {
				OR = {
					region = world_eversong_woods
					region = world_ghostlands
				}
				has_building = quel_thalas_runestone
			}
		}
		NOT = {
			OR = {
				region = world_eversong_woods
				region = world_ghostlands
			}
		}
		AND = {
			any_province = {
				OR = {
					region = world_eversong_woods
					region = world_ghostlands
				}
				has_building = quel_thalas_runestone
			}
			OR = {
				region = world_eversong_woods
				region = world_ghostlands
			}
			OR = {
				trait = high_elf
				trait = blood_elf
				trait = wretched
			}
		}
	}
}
