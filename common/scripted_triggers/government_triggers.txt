		government = troll_ke_government
		government = troll_bcd_government
		government = ogre_tribe_government
		government = helf_government
		government = noble_republic_government
	}
}

TOW_government_is_uncivilized = {
	OR = {
		TOW_is_tribal = yes
		TOW_is_creep_realm = yes
	}
}

# Replaces general condition "TOW_is_republic"
TOW_is_republic = {
	OR = {
		TOW_is_republic = yes
		government = helf_government
		government = noble_republic_government
	}
}

TOW_is_creep_realm = {
	AND = {
		OR = {
			is_nomadic = yes
			TOW_is_tribal = yes
		}
		OR = {
			AND = {
				is_ruler = yes
				TOW_is_creep_race = yes
			}
			AND = {
				is_ruler = no
				liege = {
					TOW_is_creep_race = yes
				}
			}
		}
	}
}

TOW_liege_is_head_of_kirin_tor_gov = {
	liege = {
		government = kirin_tor_government
		OR = {
			AND = {
				has_landed_title = k_dalaran
			}
			AND = {
				has_landed_title = d_dalaran
				OR = {
					k_dalaran = {
						holder_scope = {
							NOT = { government = kirin_tor_government }
						}
					}
					NOT = { is_title_active = k_dalaran }
				}
			}
			AND = {
