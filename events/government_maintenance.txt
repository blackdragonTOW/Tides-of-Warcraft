namespace = govmain

character_event = {
	id = govmain.1
	hide_window = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	trigger = {
		has_landed_title = k_kul_tiras
		culture_group = human
		any_demesne_title = {
			tier = king
			NOT = { count = 2 }
		}
		NOT = { government = noble_republic_government }
	}
	
	immediate = {
		set_government_type = noble_republic_government
	}
}
