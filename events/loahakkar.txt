#Ascend to loa on death
character_event = {
	id = loah.1
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		piety = 1000
		OR = {
			religion = loa
			religion = hakkar
		}
		NOT = { trait = race_hakkar }
		OR = {
			trait = race_troll_dark
			trait = race_troll_zandalari
			trait = race_troll_sand
			trait = race_troll_forest
			trait = race_troll_ice
			trait = race_troll_jungle
		}
	}
	
	immediate = {
		set_character_flag = ascended_loa
		current_heir = {
			if = {
				limit = {
					dynasty = ROOT
				}
				character_event = { id = loah.2 }
			}
		}
	}
}
