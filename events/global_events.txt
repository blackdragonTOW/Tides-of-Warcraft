#Call Global events
character_event = {
	id = TOW_Global.1
	hide_window = yes
	
	is_triggered_only = yes
	only_rulers = yes
	min_age = 16
	only_men = yes

	trigger = {
		NOT = { has_global_flag = TOW_global_events_called }
	}
	immediate = {
		set_global_flag = TOW_global_events_called
		log = "Global initialization events called."

		character_event = { id = TOW_Global.2 } #Province terrain additional modifiers
	}
}
