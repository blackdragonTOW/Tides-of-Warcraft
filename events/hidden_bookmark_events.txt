namespace = bookmarksetup

#This is to prevent stuff like holy light foundation and other such events from firing in startdates where they already happened.

#Also sets modifiers and flags to specific provinces.

#Sets the flags
character_event = {
	id = bookmarksetup.1
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		#Set the Bookmark global flags.
		if = {
			limit = {
				year = 7100
				NOT = { year = 9770 }
			}
			set_global_flag = trollwars_bookmark
		}
		if = {
			limit = {
				year = 9770
				NOT = { year = 10000 }
			}
			set_global_flag = threehammers_bookmark
		}
		if = {
			limit = {
				year = 10000
				NOT = { year = 10005 }
			}
			set_global_flag = firstwar_bookmark
		}
		if = {
			limit = {
				year = 10005
				NOT = { year = 10018 }
			}
			set_global_flag = secondwar_bookmark
		}
		if = {
			limit = {
				year = 10018
				NOT = { year = 10022 }
			}
			set_global_flag = thirdwar_bookmark
		}
		if = {
			limit = {
				year = 10022
				NOT = { year = 10025 }
			}
			set_global_flag = frozenthrone_bookmark
		}
		if = {
			limit = {
				year = 10025
			}
			set_global_flag = wow_bookmark
		}
		#Specific variables
		##DOOM
		set_variable = { which = global_doom value = 0 }
		#Deal with titles
		#Add province modifiers
		#Kill specific characters (usually generic scourge ones that don't have any business being around before their time)
		if = {
			limit = {
				NOT = { has_global_bookmark = wow_bookmark }
			}
			
		}
		if = {
			limit = {
				NOT = { has_global_bookmark = frozenthrone_bookmark }
			}
		}
		#Destroy settlements based on the start date
		if = {
			limit = {
				has_global_flag = threehammers_bookmark
			}
		}
		if = {
			limit = {
				OR = {
					has_global_flag = frozenthrone_bookmark
					has_global_flag = wow_bookmark
				}
			}
			b_sunstone = {
				destroy_settlemnt = THIS
			}
		}
		#Add buildings ## Due to us going past year 10k, buildings added through history won't appear (unless it was fixed)
	}
}
